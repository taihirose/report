% 課題３　閾値処理

% 閾値を4パターン設定し,閾値処理た画像を示せ．


clear; % 変数のオールクリア


ORG=imread('dog.png'); % 原画像の入力

ORG= rgb2gray(ORG); % カラー画像を白黒濃淡画像へ変換


imagesc(ORG); colormap(gray); colorbar; % 画像の表示

IMG = ORG > 64; % 輝度値が64以上の画素を1，その他を0に変換
